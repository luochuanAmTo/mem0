```py
from ollama import Client
import numpy as np

client = Client()

# memory
memory = [
    "Alex is a vegetarian",
    "Alex is allergic to nuts",
    "Ales have a cold",
    "Alex is a teacher"
]

# query
query = "What is Alex's job?"

# embedding函数
def embed(text):
    return client.embeddings(
        model="nomic-embed-text",
        prompt=text
    )["embedding"]

# 向量化
memory_vecs = [embed(m) for m in memory]
query_vec = embed(query)

# 余弦相似度
def cosine(a, b):
    return np.dot(a, b) / (np.linalg.norm(a) * np.linalg.norm(b))

print("=== Similarity ===")
for m, v in zip(memory, memory_vecs):
    print(m, "→", cosine(query_vec, v))
```

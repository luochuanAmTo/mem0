### 一 ·Episodic Memory

```python


import asyncio

from langchain.chat_models import init_chat_model
from langgraph.func import entrypoint
from langgraph.store.memory import InMemoryStore

from langmem import create_memory_store_manager

# Memory Store
store = InMemoryStore(
    index={
        "dims": 768,
        "embed": "ollama:nomic-embed-text",
    }
)

# Local Ollama LLM
llm = init_chat_model(
    "ollama:qwen3:8b"
)

# Memory manager
memory_manager = create_memory_store_manager(
    "ollama:qwen3:8b",
    namespace=("memories",),
)

# LangGraph workflow
@entrypoint(store=store)
async def chat(message: str):

    response = await llm.ainvoke(message)

    to_process = {
        "messages": [
            {"role": "user", "content": message},
            response
        ]
    }

    # 自动提取长期记忆
    await memory_manager.ainvoke(to_process)

    return response.content


async def main():

    response = await chat.ainvoke(
        "我喜欢机器人"
    )
    print("AI回复：")
    print(response)

    print("\n检索记忆：")
    memories = await store.asearch(
        ("memories",),
        query="机器人"
    )

    print(memories)


if __name__ == "__main__":
    asyncio.run(main())

```

longmem相较于mem0不是简单地提取用户喜欢机器人,而是开始自动总结Episodic Memory（情节记忆），

输入：我喜欢机器人模型自动推理用户对机器人有兴趣用户可能喜欢技术话题，用户可能对机器人应用感兴趣，Agent应该继续深入聊天

这就是：Episodic Memory（情节记忆），mem0 实际上是用户画像数据库

type=episode，即情节记忆，description是：LLM 自动生成的自然语言摘要，(p=0.95)置信度这个推断可信度 95%。 score=0.490向量相似度检索分数，表示搜索：query="机器人"这条这条 memory 的匹配程度。

### 二· LangGraph Agent + Persistent Memory

前面create_memory_store_manager()只是：自动提取 memory，下create_manage_memory_tool()升级为Agent 自主控制记忆Agent自己决定：要不要记，记什么，更新什么，删除什么

大致流程:  用户--React Agent--Tool Calling--Memory Tool--Vector Store

1.agent = create_react_agent(...)创建ReAct Agent,Reason + Act思考决定是否调用工具再继续聊天

2.create_manage_memory_tool()可以主动调用 新增记忆 更新记忆  删除记忆

3.MemorySaver(),不是长期 memory它是会话状态保存器用于：保存聊天历史,保存graph执行状态,即当前上下文，

4.store = InMemoryStore()：长期记忆库，memories = store.search(...)从记忆库查询真正实现了回忆

```
from langgraph.checkpoint.memory import MemorySaver
from langgraph.prebuilt import create_react_agent
from langgraph.store.memory import InMemoryStore
from langgraph.utils.config import get_store

from langmem import create_manage_memory_tool

# 长期记忆库
store = InMemoryStore(
    index={
        "dims": 768,
        "embed": "ollama:nomic-embed-text",
    }
)

# 会话状态
checkpointer = MemorySaver()

# prompt函数
def prompt(state):

    store_instance = get_store()

    memories = store_instance.search(
        ("memories",),
        query=state["messages"][-1].content,
    )

    system_msg = f"""
You are a helpful assistant.

## Memories
{memories}
"""

    return [
        {"role": "system", "content": system_msg},
        *state["messages"]
    ]


# 创建Agent
agent = create_react_agent(
    "ollama:qwen3:8b",

    prompt=prompt,

    tools=[
        create_manage_memory_tool(
            namespace=("memories",)
        )
    ],

    store=store,

    checkpointer=checkpointer,
)

# 会话1
config = {
    "configurable": {
        "thread_id": "thread-a"
    }
}

response = agent.invoke(
    {
        "messages": [
            {
                "role": "user",
                "content": "我喜欢机器人，记住这个"
            }
        ]
    },
    config=config,
)

print(response["messages"][-1].content)

# 新会话
new_config = {
    "configurable": {
        "thread_id": "thread-b"
    }
}

response2 = agent.invoke(
    {
        "messages": [
            {
                "role": "user",
                "content": "你记得我喜欢什么吗？"
            }
        ]
    },
    config=new_config,
)

print(response2["messages"][-1].content)
```
